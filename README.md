## Usage

```js
import tooltip from '@wtjs/tooltip';

<Tooltip render="Hi, I'm a tooltip">
    Hover me
</Tooltip>

<Tooltip render={() => <div>Hi I am a tooltip</div>}>
    Hover me
</Tooltip>
```
